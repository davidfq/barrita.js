barrita.js

jQuery (and Zepto) plugin to create progress bars, heavily inspired by nprogress

see code and documentation at github

show code 


var barrita = $("#demo_start-stop").barrita({ 
		autoStart : false, 
		hideOnDone : false 
	}).data('barrita');

barrita.start();
barrita.stop();
barrita.done();

var options = {
	// default value
	autoStart : true, 
	// default value
	hideOnDone : true, 
	// time between increments (millis)
	speed : 2500, 
	// transition speed (millis)
	trickleRate : 500 
};
$('#demo_settings').barrita(options);
